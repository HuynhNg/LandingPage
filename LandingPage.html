// golang

// Get unique values from the array of int values 

func unique(arr []int) []int {
    seen := make(map[int]bool)
    uniques := make([]int, 0)
    for _, v := range arr {
        if _, ok := seen[v]; !ok {
            seen[v] = true
            uniques = append(uniques, v)
        }
    }
    return uniques
}
